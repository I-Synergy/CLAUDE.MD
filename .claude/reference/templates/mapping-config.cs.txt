// File: {ApplicationName}.Domain.{Domain}/Mappers/{Entity}MappingConfig.cs

namespace {ApplicationName}.Domain.{Domain}.Mappers;

using Mapster;

/// <summary>
/// Mapster configuration for {Entity} mappings.
/// </summary>
public class {Entity}MappingConfig : IRegister
{
    /// <summary>
    /// Registers all mapping configurations.
    /// </summary>
    public void Register(TypeAdapterConfig config)
    {
        // Command -> Entity
        config.NewConfig<Create{Entity}Command, {Entity}>()
            .Map(dest => dest.Property1, src => src.Property1)
            .Map(dest => dest.Property2, src => src.Property2)
            .Map(dest => dest.Property3, src => src.Property3)
            .Map(dest => dest.{Entity}Id, src => Guid.NewGuid())
            .Map(dest => dest.CreatedDate, src => DateTimeOffset.UtcNow);

        config.NewConfig<Update{Entity}Command, {Entity}>()
            .Map(dest => dest.Property1, src => src.Property1)
            .Map(dest => dest.Property2, src => src.Property2)
            .Map(dest => dest.ChangedDate, src => DateTimeOffset.UtcNow);

        // Entity -> Model
        config.NewConfig<{Entity}, {Entity}Model>()
            .Map(dest => dest.{Entity}Id, src => src.{Entity}Id)
            .Map(dest => dest.Property1, src => src.Property1)
            .Map(dest => dest.Property2, src => src.Property2)
            .Map(dest => dest.Property3, src => src.Property3)
            .Map(dest => dest.CreatedDate, src => src.CreatedDate)
            .Map(dest => dest.ChangedDate, src => src.ChangedDate);

        // Model -> Entity
        config.NewConfig<{Entity}Model, {Entity}>()
            .Map(dest => dest.{Entity}Id, src => src.{Entity}Id)
            .Map(dest => dest.Property1, src => src.Property1)
            .Map(dest => dest.Property2, src => src.Property2)
            .Map(dest => dest.Property3, src => src.Property3)
            .Map(dest => dest.CreatedDate, src => src.CreatedDate)
            .Map(dest => dest.ChangedDate, src => src.ChangedDate);

        // Model -> Response
        config.NewConfig<{Entity}Model, {Entity}Response>()
            .Map(dest => dest.{Entity}Id, src => src.{Entity}Id)
            .Map(dest => dest.Property1, src => src.Property1)
            .Map(dest => dest.Property2, src => src.Property2)
            .Map(dest => dest.Property3, src => src.Property3)
            .Map(dest => dest.CreatedDate, src => src.CreatedDate)
            .Map(dest => dest.ChangedDate, src => src.ChangedDate);
    }
}
